import { MetadataRoute } from 'next';

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: '*', // 모든 검색 로봇에게 허용
      allow: '/',     // 모든 페이지 수집 허용
      disallow: [
        '/admin/',    // 관리자 페이지는 수집 금지
        '/api/',      // API 경로는 수집 금지
        '/saved', '/login', '/register','/chat/[id]', '/auth/',
      ],
    },
    sitemap: 'https://mappyenglish.com/sitemap.xml', // 사이트맵 위치 알려주기
  };
}
